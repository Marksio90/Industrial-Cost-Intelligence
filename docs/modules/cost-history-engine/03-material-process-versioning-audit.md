# Section 7 — Material Price History

Material price history is the foundational time-series layer of the Cost History Engine. Every external commodity index feed, contract price, and internally validated spot rate is persisted as an immutable `MaterialPriceRecord`. Forecasts are generated by the `PriceForecastEngine` and stored in `ForecastRecord`. Daily batch jobs compute rolling index metrics into `MaterialPriceIndex`.

---

## MaterialPriceRecord Entity

Primary storage for a single price observation. One row per source event per material per date. Records are **never updated**; corrections create a new row with `version + 1` and set `superseded_by` on the old row.

| Column | Type | Constraints | Description |
|---|---|---|---|
| `record_id` | `UUID` | PK DEFAULT `gen_random_uuid()` | Surrogate primary key |
| `material_id` | `UUID` | NOT NULL · FK → MIE `materials.material_id` | Reference to material master in MIE |
| `material_grade` | `VARCHAR(50)` | — | Specific alloy/grade string, e.g. `EN AW-6082-T6`, `S235JR` |
| `price_per_kg_eur` | `NUMERIC(18,6)` | — | Derived price per kilogram in EUR; NULL if unit is not weight-based |
| `price_per_unit_eur` | `NUMERIC(18,6)` | NOT NULL | Price in EUR per the record's unit |
| `unit` | `VARCHAR(20)` | NOT NULL · CHECK IN (`KG`,`TONNE`,`METER`,`PIECE`) | Physical unit of measure |
| `source` | `ENUM` | NOT NULL | One of `MARKET_INDEX`, `SPOT`, `CONTRACT`, `ESTIMATED`, `ACTUAL` |
| `index_name` | `VARCHAR(50)` | — | Index identifier, e.g. `LME_ALUMINUM`, `PLATTS_HRC_EU` |
| `raw_value` | `NUMERIC(18,6)` | — | Value as received from source, in `currency_original` and original unit |

## PriceForecastEngine

Multi-model price forecast engine. Implements `LINEAR` regression natively; `PROPHET` and `ARIMA` are integrated via optional dependencies. `ENSEMBLE` produces a weighted average of all fitted models. Design target: MAPE < 8% on a 90-day walk-forward backtest.

Forecast outputs are persisted as `ForecastRecord` rows (see entity above). The `BacktestResult.passed` flag gates automatic publication of forecast values to the `che.material-prices.forecasts` Kafka topic.

```python
from dataclasses import dataclass, field
from decimal import Decimal
from datetime import date, datetime, timedelta
from enum import Enum
from typing import Optional
import uuid
import statistics
import math


class ModelType(str, Enum):
    PROPHET = "PROPHET"
    ARIMA = "ARIMA"
    LINEAR = "LINEAR"
    ENSEMBLE = "ENSEMBLE"


@dataclass
class ForecastResult:
    material_id: uuid.UUID
    model_type: ModelType
    forecast_date: date
    forecast_value_eur: Decimal
    confidence_lower: Decimal
    confidence_upper: Decimal
    mape_last_backtest: Optional[Decimal]
    model_version: str
    horizon_days: int
    feature_set: dict = field(default_factory=dict)


@dataclass
class BacktestResult:
    model_type: ModelType
    mape: Decimal
    mae: Decimal
    rmse: Decimal
    n_periods: int
    start_date: date
    end_date: date
    passed: bool  # True if MAPE < 8%


MAPE_TARGET = Decimal("8.0")  # Maximum acceptable MAPE percentage


class PriceForecastEngine:
    """
    Multi-model price forecast engine for material price history.

    Supports LINEAR (built-in), PROPHET/ARIMA (via optional dependencies),
    and ENSEMBLE (weighted average of available models).

    Target: MAPE < 8% on 90-day walk-forward backtest (see BacktestResult.passed).
    Forecast outputs are persisted as ForecastRecord rows by the caller.
    """

    MODEL_VERSION = "1.2.0"

    def __init__(self, material_id: uuid.UUID) -> None:
        self.material_id = material_id
        self._price_series: list[tuple[date, Decimal]] = []
        self._fitted_models: dict[ModelType, object] = {}

    def fit(self, price_series: list[tuple[date, Decimal]]) -> "PriceForecastEngine":
        """Load and sort historical price series for model fitting."""
        if len(price_series) < 12:
            raise ValueError(
                f"Insufficient data: need ≥12 data points, got {len(price_series)}."
            )
        self._price_series = sorted(price_series, key=lambda x: x[0])
        self._fit_linear()
        return self

    def _fit_linear(self) -> None:
        """Fit simple OLS linear regression on time index → price."""
        prices = [float(p) for _, p in self._price_series]
        n = len(prices)
        x_vals = list(range(n))
        x_mean = sum(x_vals) / n
        y_mean = sum(prices) / n
        numerator = sum(
            (x - x_mean) * (y - y_mean) for x, y in zip(x_vals, prices)
        )
        denominator = sum((x - x_mean) ** 2 for x in x_vals)
        slope = numerator / denominator if denominator != 0 else 0.0
        intercept = y_mean - slope * x_mean
        self._fitted_models[ModelType.LINEAR] = {
            "slope": slope,
            "intercept": intercept,
            "n": n,
        }

    def _predict_linear(self, steps_ahead: int) -> Decimal:
        params = self._fitted_models[ModelType.LINEAR]
        value = params["intercept"] + params["slope"] * (params["n"] - 1 + steps_ahead)
        return Decimal(str(round(max(value, 0.0), 6)))

    def _compute_confidence_interval(
        self, prediction: Decimal, horizon_days: int
    ) -> tuple[Decimal, Decimal]:
        """
        Approximate 80% CI: ±1.28σ, where σ grows with sqrt(horizon / 30).
        Uses residual standard deviation from the fitted linear model.
        """
        prices = [float(p) for _, p in self._price_series]
        params = self._fitted_models[ModelType.LINEAR]
        residuals = [
            prices[i] - (params["intercept"] + params["slope"] * i)
            for i in range(len(prices))
        ]
        residual_std = statistics.stdev(residuals) if len(residuals) > 1 else 0.0
        ci_half = 1.28 * residual_std * math.sqrt(horizon_days / 30)
        lower = Decimal(str(round(max(float(prediction) - ci_half, 0.0), 6)))
        upper = Decimal(str(round(float(prediction) + ci_half, 6)))
        return lower, upper

    def predict(
        self,
        target_date: date,
        model_type: ModelType = ModelType.LINEAR,
    ) -> ForecastResult:
        """Generate a single-date point forecast with 80% confidence interval."""
        if not self._price_series:
            raise RuntimeError("Call fit() before predict().")
        last_date = self._price_series[-1][0]
        horizon_days = (target_date - last_date).days
        if horizon_days <= 0:
            raise ValueError(
                f"Target date {target_date} must be after last data point {last_date}."
            )

        if model_type in (ModelType.LINEAR, ModelType.ENSEMBLE):
            value = self._predict_linear(horizon_days)
        else:
            raise NotImplementedError(
                f"Model {model_type} requires an external dependency "
                f"(prophet / statsmodels). Fit that model first."
            )

        lower, upper = self._compute_confidence_interval(value, horizon_days)
        return ForecastResult(
            material_id=self.material_id,
            model_type=model_type,
            forecast_date=target_date,
            forecast_value_eur=value,
            confidence_lower=lower,
            confidence_upper=upper,
            mape_last_backtest=None,  # Populated after calling backtest()
            model_version=self.MODEL_VERSION,
            horizon_days=horizon_days,
        )

    def ensemble_forecast(self, target_date: date) -> ForecastResult:
        """
        Weighted ensemble of all fitted models.

        Weights are inversely proportional to backtest MAPE when available;
        falls back to equal weights. Degrades gracefully to a single-model
        forecast when only one model is fitted.
        """
        available = list(self._fitted_models.keys())
        if len(available) == 1:
            return self.predict(target_date, available[0])

        predictions = {m: self.predict(target_date, m) for m in available}
        total_weight = len(available)
        ensemble_value = (
            sum(float(r.forecast_value_eur) for r in predictions.values())
            / total_weight
        )

        all_lowers = [float(r.confidence_lower) for r in predictions.values()]
        all_uppers = [float(r.confidence_upper) for r in predictions.values()]
        last_date = self._price_series[-1][0]
        horizon_days = (target_date - last_date).days

        return ForecastResult(
            material_id=self.material_id,
            model_type=ModelType.ENSEMBLE,
            forecast_date=target_date,
            forecast_value_eur=Decimal(str(round(ensemble_value, 6))),
            confidence_lower=Decimal(str(round(min(all_lowers), 6))),
            confidence_upper=Decimal(str(round(max(all_uppers), 6))),
            mape_last_backtest=None,
            model_version=self.MODEL_VERSION,
            horizon_days=horizon_days,
        )

    def backtest(
        self,
        holdout_periods: int = 3,
        model_type: ModelType = ModelType.LINEAR,
    ) -> BacktestResult:
        """
        Walk-forward backtest: fit on all-but-last N periods, predict each
        held-out period. Returns BacktestResult with MAPE, MAE, RMSE and a
        boolean pass/fail flag against the 8% MAPE target.
        """
        if len(self._price_series) < holdout_periods + 12:
            raise ValueError(
                f"Need ≥{holdout_periods + 12} data points for backtest "
                f"with {holdout_periods} holdout periods."
            )
        train_series = self._price_series[:-holdout_periods]
        test_series = self._price_series[-holdout_periods:]

        test_engine = PriceForecastEngine(self.material_id)
        test_engine.fit(train_series)

        actuals = [float(p) for _, p in test_series]
        predicted = [
            float(test_engine.predict(target_date, model_type).forecast_value_eur)
            for target_date, _ in test_series
        ]

        mape_values = [
            abs(a - p) / abs(a) * 100
            for a, p in zip(actuals, predicted)
            if a != 0
        ]
        mape = (
            Decimal(str(round(statistics.mean(mape_values), 4)))
            if mape_values
            else Decimal("0")
        )
        mae_val = statistics.mean([abs(a - p) for a, p in zip(actuals, predicted)])
        rmse_val = math.sqrt(
            statistics.mean([(a - p) ** 2 for a, p in zip(actuals, predicted)])
        )

        return BacktestResult(
            model_type=model_type,
            mape=mape,
            mae=Decimal(str(round(mae_val, 6))),
            rmse=Decimal(str(round(rmse_val, 6))),
            n_periods=holdout_periods,
            start_date=test_series[0][0],
            end_date=test_series[-1][0],
            passed=mape <= MAPE_TARGET,
        )
```

# Section 8 — Process Cost History

Process cost history captures the time-series of manufacturing hourly rates at machine and plant level. Records feed the Cost Comparison Engine (CCE) and the Quote pricing engine. OEE (Overall Equipment Effectiveness) data is stored as a companion snapshot for each cost period to enable efficiency-adjusted cost analysis.

> **Cross-reference:** `process_code` values (`CUT`, `MAC`, `FOR`, `JOI`, `ASS`, `FIN`) are the canonical MPE process codes defined in the Manufacturing Process Engine schema. `machine_id` and `plant_id` are foreign keys into MPE.

---

One row per process type / machine / plant / period / version. Records are **immutable**; cost revisions create a new row with `version + 1` and set `superseded_by` on the old row. The `total_hourly_rate_eur` column is a PostgreSQL 16 generated column.

```python
from enum import Enum

class ProcessCostTrendAnalyzer:
    """
    Analyses process cost history: evolution trends, cost driver identification,
    and benchmarking against industry survey data.

    All methods are stateless and accept pre-fetched record dicts so the class
    can be unit-tested without a database connection.
    """

    def calculate_cost_evolution(
        self,
        records: list[dict],
        # Each dict: {"period_year": int, "period_month": int,
        #             "total_hourly_rate_eur": Decimal, "record_id": UUID}
    ) -> CostEvolution:
        """
        Calculate hourly rate evolution over time, year-over-year change
        percentage (requires ≥13 months of data), trend direction and
        month-over-month volatility.
        """
        sorted_records = sorted(
            records, key=lambda r: (r["period_year"], r["period_month"])
        )
        periods = [
            f"{r['period_year']}-{r['period_month']:02d}" for r in sorted_records
        ]
        rates = [r["total_hourly_rate_eur"] for r in sorted_records]
        record_ids = [r["record_id"] for r in sorted_records]

        yoy_change_pct: Optional[Decimal] = None
        if len(rates) >= 13:
            current = rates[-1]
            prior_year = rates[-13]
            if prior_year != 0:
                yoy_change_pct = (
                    (current - prior_year) / prior_year * Decimal("100")
                ).quantize(Decimal("0.01"))

        monthly_changes = [
            float(rates[i] - rates[i - 1]) for i in range(1, len(rates))
        ]
        volatility = (
            Decimal(str(round(statistics.stdev(monthly_changes), 6)))
            if len(monthly_changes) > 1
            else Decimal("0")
        )

        if len(rates) >= 2:
            net_change = rates[-1] - rates[0]
            if net_change > Decimal("0.5"):
                trend = "INCREASING"
            elif net_change < Decimal("-0.5"):
                trend = "DECREASING"
            else:
                trend = "STABLE"
        else:
            trend = "STABLE"

        return CostEvolution(
            record_ids=record_ids,
            periods=periods,
            hourly_rates=rates,
            yoy_change_pct=yoy_change_pct,
            trend_direction=trend,
            volatility=volatility,
        )

    def identify_cost_drivers(self, component: dict) -> CostDriverAnalysis:
        """
        Identify the dominant cost driver from a CostComponent record dict.

        component keys: record_id, labor_pct, energy_pct, depreciation_pct,
                        maintenance_pct, overhead_pct (all Decimal).
        """
        drivers: dict[str, Decimal] = {
            "LABOR": component.get("labor_pct", Decimal("0")),
            "ENERGY": component.get("energy_pct", Decimal("0")),
            "DEPRECIATION": component.get("depreciation_pct", Decimal("0")),
            "MAINTENANCE": component.get("maintenance_pct", Decimal("0")),
            "OVERHEAD": component.get("overhead_pct", Decimal("0")),
        }
        dominant = max(drivers, key=lambda k: float(drivers[k]))
        dominant_pct = drivers[dominant]

        insight_map: dict[str, str] = {
            "LABOR": (
                "Labor-intensive process. Evaluate automation ROI and operator "
                "multi-skilling to reduce direct labor allocation."
            ),
            "ENERGY": (
                "Energy-intensive process. Review energy tariff structure, "
                "off-peak scheduling and equipment efficiency ratings."
            ),
            "DEPRECIATION": (
                "High capital cost allocation. Validate machine utilization rate "
                "via OEE and consider shared-asset charging model."
            ),
            "MAINTENANCE": (
                "High maintenance burden. Review preventive maintenance schedule, "
                "spare-parts inventory strategy and MTBF trends."
            ),
            "OVERHEAD": (
                "High overhead allocation. Review cost-centre allocation keys "
                "and validate indirect headcount attribution methodology."
            ),
        }

        return CostDriverAnalysis(
            record_id=component["record_id"],
            dominant_driver=dominant,
            driver_pct=dominant_pct,
            all_drivers=drivers,
            insight=insight_map[dominant],
        )

    def benchmark_vs_industry(
        self,
        record_id: uuid.UUID,
        process_code: ProcessCode,
        plant_rate_eur_hr: Decimal,
        region: BenchmarkRegion,
        benchmark_data: dict,
        # {"min": Decimal, "median": Decimal, "max": Decimal}
    ) -> BenchmarkComparison:
        """
        Compare plant hourly rate against industry benchmark for the given
        process class and region.

        Position thresholds (vs survey median):
          ≤ -10%  → BEST_IN_CLASS
          ≤  +5%  → COMPETITIVE
          ≤ +25%  → ABOVE_MEDIAN
          >  +25% → HIGH_COST
        """
        benchmark_median = benchmark_data["median"]
        vs_median_pct = (
            (plant_rate_eur_hr - benchmark_median) / benchmark_median * Decimal("100")
        ).quantize(Decimal("0.01"))

        if vs_median_pct <= Decimal("-10"):
            position = "BEST_IN_CLASS"
        elif vs_median_pct <= Decimal("5"):
            position = "COMPETITIVE"
        elif vs_median_pct <= Decimal("25"):
            position = "ABOVE_MEDIAN"
        else:
            position = "HIGH_COST"

        return BenchmarkComparison(
            record_id=record_id,
            process_code=process_code,
            plant_rate_eur_hr=plant_rate_eur_hr,
            benchmark_median_eur_hr=benchmark_median,
            benchmark_min_eur_hr=benchmark_data["min"],
            benchmark_max_eur_hr=benchmark_data["max"],
            vs_median_pct=vs_median_pct,
            position=position,
        )
```

# Section 9 — Versioning

The CHE versioning subsystem enforces a strict **immutable-record** discipline: no core cost or price row is ever mutated after creation. All corrections and state transitions produce new database rows. This section defines the versioning strategy, the supporting entities, and the `VersionManager` service class.

> **Cross-reference:** The `entity_type` values used in `EntityVersion` correspond to the entity names defined throughout this document and in `02-cost-snapshot-quoting.md` (e.g. `COST_SNAPSHOT`, `QUOTE`, `RFQ_BID`, `SUPPLIER_PRICE_RECORD`).

---

## Versioning Strategy Overview

### 1. Optimistic Locking via `version` Field

Every core entity (`MaterialPriceRecord`, `ProcessCostRecord`, `CostSnapshot`, etc.) carries a `version INT NOT NULL DEFAULT 1` column. The application layer includes `WHERE version = :expected_version` on every logical update operation. If the row was modified concurrently the affected-row count returns 0, triggering a `ConcurrentModificationError` and prompting the caller to re-fetch and retry.

### 2. Semantic Versioning for Snapshots

Snapshot-level versions follow **MAJOR.MINOR.PATCH** semantics:

| Increment | Trigger condition |
|---|---|
| `MAJOR` | Breaking cost change: total cost delta > 10% vs prior version |
| `MINOR` | Line-item addition, removal or change; workflow state transition (APPROVED, SUPERSEDED) |
| `PATCH` | Metadata correction: tags, notes, labels, non-cost fields |

### 3. Immutable Records

No `UPDATE` statement is issued on core entity tables by the CHE application layer. The sequence for any correction is:

1. `INSERT` a new row with `version = old.version + 1` and corrected field values.
2. `UPDATE old_row SET superseded_by = new_row.record_id` — the only permitted UPDATE, restricted to the `superseded_by` column by a column-level permission grant.

### 4. Superseded-by Chain

Each entity row carries a `superseded_by UUID FK SELF` column. Following the chain from any record to a `NULL` `superseded_by` value reaches the **current active record**. The chain is always a linear linked list (no branching); parallel corrections are serialised through optimistic locking.

### 5. `is_current` Flag

`EntityVersion.is_current BOOLEAN` provides O(1) lookup of the active version without traversing the superseded-by chain. It is set to `FALSE` on all prior versions and `TRUE` on the new version within the same DB transaction that performs the superseded-by update.

---

## VersionManager

Manages the complete version lifecycle for CHE entities. All write operations produce **new** `EntityVersionRecord` objects; the caller is responsible for persisting them to the database within a transaction that also updates `is_current` flags and `superseded_by` pointers.

```python
from dataclasses import dataclass, field
from datetime import datetime
from decimal import Decimal
from enum import Enum
from typing import Any, Optional
import uuid
import json


class ChangeType(str, Enum):
    CREATED = "CREATED"
    UPDATED = "UPDATED"
    APPROVED = "APPROVED"
    SUPERSEDED = "SUPERSEDED"
    ARCHIVED = "ARCHIVED"


class FieldSensitivity(str, Enum):
    PUBLIC = "PUBLIC"
    INTERNAL = "INTERNAL"
    CONFIDENTIAL = "CONFIDENTIAL"
    RESTRICTED = "RESTRICTED"


# Fields whose values must not appear in the unencrypted diff_jsonb column
SENSITIVE_FIELDS: frozenset[str] = frozenset({
    "floor_price_eur",
    "margin_pct",
    "profit_margin_pct",
    "target_price_eur",
    "bid_price",
    "bid_price_eur",
    "price_per_unit",
    "price_eur_normalized",
})


@dataclass
class VersionDiffEntry:
    field_name: str
    old_value: Any
    new_value: Any
    delta_absolute: Optional[Decimal] = None
    delta_pct: Optional[Decimal] = None
    field_sensitivity: FieldSensitivity = FieldSensitivity.INTERNAL


@dataclass
class EntityVersionRecord:
    version_id: uuid.UUID
    entity_type: str
    entity_id: uuid.UUID
    version_number: int
    semantic_version: str
    change_type: ChangeType
    change_summary: str
    diff_jsonb: dict
    created_by: uuid.UUID
    created_at: datetime
    is_current: bool = True
    diffs: list[VersionDiffEntry] = field(default_factory=list)


@dataclass
class VersionLabel:
    label_id: uuid.UUID
    entity_type: str
    entity_id: uuid.UUID
    version_id: uuid.UUID
    label_name: str
    description: str
    labeled_by: uuid.UUID
    labeled_at: datetime = field(default_factory=datetime.utcnow)


class VersionManager:
    """
    Manages version lifecycle for CHE entities.

    All mutations produce new EntityVersionRecord instances.
    Existing version records are never mutated.
    Caller persists returned records to the database within a transaction
    that also flips is_current flags and sets superseded_by pointers.
    """

    def _compute_semantic_version(
        self,
        prior_semantic: str,
        old_values: dict,
        new_values: dict,
        change_type: ChangeType,
    ) -> str:
        """Determine MAJOR.MINOR.PATCH increment based on change magnitude and type."""
        major, minor, patch = [int(x) for x in prior_semantic.split(".")]

        if change_type in (
            ChangeType.APPROVED,
            ChangeType.SUPERSEDED,
            ChangeType.ARCHIVED,
        ):
            return f"{major}.{minor + 1}.0"

        # Major bump: total cost delta > 10%
        old_cost = old_values.get("total_cost_eur") or old_values.get("total_price_eur")
        new_cost = new_values.get("total_cost_eur") or new_values.get("total_price_eur")
        if old_cost is not None and new_cost is not None:
            try:
                delta_pct = abs(
                    (Decimal(str(new_cost)) - Decimal(str(old_cost)))
                    / Decimal(str(old_cost))
                    * 100
                )
                if delta_pct > Decimal("10"):
                    return f"{major + 1}.0.0"
            except Exception:
                pass

        # Minor bump: line-item changes
        if "line_items" in new_values or "quote_lines" in new_values:
            return f"{major}.{minor + 1}.0"

        # Patch bump: metadata / tag / minor field corrections
        return f"{major}.{minor}.{patch + 1}"

    def _compute_diff(
        self, old_values: dict, new_values: dict
    ) -> list[VersionDiffEntry]:
        """Generate a per-field diff between old and new entity state."""
        diffs: list[VersionDiffEntry] = []
        all_fields = sorted(set(old_values.keys()) | set(new_values.keys()))

        for field_name in all_fields:
            old_val = old_values.get(field_name)
            new_val = new_values.get(field_name)
            if old_val == new_val:
                continue

            sensitivity = (
                FieldSensitivity.RESTRICTED
                if field_name in SENSITIVE_FIELDS
                else FieldSensitivity.INTERNAL
            )
            delta_abs: Optional[Decimal] = None
            delta_pct_val: Optional[Decimal] = None
            try:
                old_d = Decimal(str(old_val)) if old_val is not None else None
                new_d = Decimal(str(new_val)) if new_val is not None else None
                if old_d is not None and new_d is not None:
                    delta_abs = (new_d - old_d).quantize(Decimal("0.000001"))
                    if old_d != 0:
                        delta_pct_val = (delta_abs / old_d * 100).quantize(
                            Decimal("0.01")
                        )
            except Exception:
                pass

            diffs.append(
                VersionDiffEntry(
                    field_name=field_name,
                    old_value=old_val,
                    new_value=new_val,
                    delta_absolute=delta_abs,
                    delta_pct=delta_pct_val,
                    field_sensitivity=sensitivity,
                )
            )
        return diffs

    def create_version(
        self,
        entity_type: str,
        entity_id: uuid.UUID,
        prior_version_number: int,
        prior_semantic_version: str,
        change_type: ChangeType,
        change_summary: str,
        old_values: dict,
        new_values: dict,
        created_by: uuid.UUID,
    ) -> EntityVersionRecord:
        """
        Create and return a new EntityVersionRecord.

        RESTRICTED-sensitivity fields are excluded from diff_jsonb but
        are included in the returned diffs list for the caller to persist
        encrypted in the VersionDiff table.
        """
        new_version_number = prior_version_number + 1
        new_semantic = self._compute_semantic_version(
            prior_semantic_version, old_values, new_values, change_type
        )
        diffs = self._compute_diff(old_values, new_values)

        # Build diff_jsonb excluding RESTRICTED fields (encrypted separately)
        diff_jsonb = {
            d.field_name: {
                "old": d.old_value,
                "new": d.new_value,
                "delta_abs": str(d.delta_absolute) if d.delta_absolute is not None else None,
                "delta_pct": str(d.delta_pct) if d.delta_pct is not None else None,
            }
            for d in diffs
            if d.field_sensitivity is not FieldSensitivity.RESTRICTED
        }

        return EntityVersionRecord(
            version_id=uuid.uuid4(),
            entity_type=entity_type,
            entity_id=entity_id,
            version_number=new_version_number,
            semantic_version=new_semantic,
            change_type=change_type,
            change_summary=change_summary,
            diff_jsonb=diff_jsonb,
            created_by=created_by,
            created_at=datetime.utcnow(),
            is_current=True,
            diffs=diffs,
        )

    def get_version_at(
        self,
        version_history: list[EntityVersionRecord],
        point_in_time: datetime,
    ) -> Optional[EntityVersionRecord]:
        """Return the version that was current at a given point in time."""
        candidates = [v for v in version_history if v.created_at <= point_in_time]
        if not candidates:
            return None
        return max(candidates, key=lambda v: v.created_at)

    def diff_versions(
        self,
        version_a: EntityVersionRecord,
        version_b: EntityVersionRecord,
    ) -> list[VersionDiffEntry]:
        """Compute the diff between two arbitrary versions by comparing their stored new values."""
        fields_a = {
            field_name: data.get("new")
            for field_name, data in version_a.diff_jsonb.items()
        }
        fields_b = {
            field_name: data.get("new")
            for field_name, data in version_b.diff_jsonb.items()
        }
        return self._compute_diff(fields_a, fields_b)

    def rollback_to(
        self,
        entity_type: str,
        entity_id: uuid.UUID,
        target_version: EntityVersionRecord,
        current_values: dict,
        rolled_back_by: uuid.UUID,
        reason: str,
    ) -> EntityVersionRecord:
        """
        Rollback by creating a NEW version that restores the state captured
        in target_version. Does NOT mutate any existing EntityVersionRecord.
        """
        old_state = {
            field_name: data.get("old")
            for field_name, data in target_version.diff_jsonb.items()
        }
        return self.create_version(
            entity_type=entity_type,
            entity_id=entity_id,
            prior_version_number=target_version.version_number,
            prior_semantic_version=target_version.semantic_version,
            change_type=ChangeType.UPDATED,
            change_summary=(
                f"Rollback to v{target_version.semantic_version} — {reason}"
            ),
            old_values=current_values,
            new_values=old_state,
            created_by=rolled_back_by,
        )

    def label_version(
        self,
        entity_type: str,
        entity_id: uuid.UUID,
        version_id: uuid.UUID,
        label_name: str,
        description: str,
        labeled_by: uuid.UUID,
    ) -> VersionLabel:
        """Create a named label checkpoint for a specific entity version."""
        return VersionLabel(
            label_id=uuid.uuid4(),
            entity_type=entity_type,
            entity_id=entity_id,
            version_id=version_id,
            label_name=label_name,
            description=description,
            labeled_by=labeled_by,
        )
```

## Snapshot Lineage Graph

Each snapshot entity carries a `superseded_by` pointer to the next version in the correction chain, forming a directed acyclic graph (linear chain). The `SnapshotLineageTracer` walks this chain to reconstruct the full history of a snapshot from its original creation through to the currently active version.

### Example Lineage Chain

```
┌───────────────────────────────────┐
│  snapshot_id: a1b2-...(v1)        │
│  snapshot_type: INITIAL_ESTIMATE  │
│  version: 1                       │
│  status: SUPERSEDED               │
│  superseded_by: c3d4-...          │
└───────────────┬───────────────────┘
                │
                ▼
┌───────────────────────────────────┐
│  snapshot_id: c3d4-...(v2)        │
│  snapshot_type: REVISED           │
│  version: 2                       │
│  status: SUPERSEDED               │
│  superseded_by: e5f6-...          │
└───────────────┬───────────────────┘
                │
                ▼
┌───────────────────────────────────┐
│  snapshot_id: e5f6-...(v3)        │
│  snapshot_type: REVISED           │
│  version: 3                       │
│  status: SUPERSEDED               │
│  superseded_by: g7h8-...          │
└───────────────┬───────────────────┘
                │
                ▼
┌───────────────────────────────────┐
│  snapshot_id: g7h8-...(v4)        │
│  snapshot_type: APPROVED          │
│  version: 4                       │
│  status: APPROVED                 │
│  superseded_by: NULL  ◄─ CURRENT  │
└───────────────────────────────────┘
```

### SnapshotLineageTracer

```python
from dataclasses import dataclass, field
from typing import Optional
import uuid


@dataclass
class LineageNode:
    snapshot_id: uuid.UUID
    version: int
    snapshot_type: str
    status: str
    parent_snapshot_id: Optional[uuid.UUID]
    children: list["LineageNode"] = field(default_factory=list)


class SnapshotLineageTracer:
    """
    Traces the lineage chain of a snapshot from the root (original creation)
    to the latest version by following the superseded_by chain.

    all_snapshots schema:
        {
          snapshot_id (UUID): {
            "version": int,
            "snapshot_type": str,
            "status": str,
            "superseded_by": UUID | None
          }
        }
    """

    def trace(
        self,
        target_snapshot_id: uuid.UUID,
        all_snapshots: dict[uuid.UUID, dict],
    ) -> LineageNode:
        """
        Build a lineage tree rooted at the original snapshot.
        Each node's `children` list contains the snapshots it was superseded by.
        """
        # Build a reverse index: child_id -> parent_id
        # A snapshot is a "parent" of the snapshot it points to via superseded_by
        parent_of: dict[uuid.UUID, uuid.UUID] = {}
        for snap_id, snap in all_snapshots.items():
            superseded_by = snap.get("superseded_by")
            if superseded_by and superseded_by in all_snapshots:
                parent_of[superseded_by] = snap_id

        # Walk back to root: the snapshot with no parent in parent_of
        current = target_snapshot_id
        visited: set[uuid.UUID] = set()
        while current in parent_of and parent_of[current] not in visited:
            visited.add(current)
            current = parent_of[current]
        root_id = current

        def build_node(snap_id: uuid.UUID) -> LineageNode:
            snap = all_snapshots[snap_id]
            node = LineageNode(
                snapshot_id=snap_id,
                version=snap["version"],
                snapshot_type=snap["snapshot_type"],
                status=snap["status"],
                parent_snapshot_id=parent_of.get(snap_id),
            )
            # Follow superseded_by forward to build the child chain
            child_id = snap.get("superseded_by")
            if child_id and child_id in all_snapshots:
                node.children.append(build_node(child_id))
            return node

        return build_node(root_id)
```

## Version Retention Policy

| Change Type | Retention Rule | Responsible Job |
|---|---|---|
| `APPROVED` | Retained indefinitely; exempt from all purge cycles | — |
| `DRAFT` | Purged 90 days after creation if still in DRAFT status | `DraftVersionPurgeJob` |
| `SUPERSEDED` | Retained for 7 years from supersession date (EU commercial law) | `RetentionPolicyExecution` |
| `ARCHIVED` | Retained for 7 years from archive date, then deleted with full audit trail entry | `RetentionPolicyExecution` |

The `RetentionPolicyExecution` Kafka event is consumed by `AuditEventConsumer` (Section 10) and recorded as an `ARCHIVE` audit action to provide a complete deletion audit trail.

# Section 10 — Audit Trail

The CHE audit trail is an **append-only, tamper-evident** log of every state-changing and sensitive data-reading operation across all CHE entities. It satisfies EU GDPR Article 30 (processing records), German HGB § 257 (7-year commercial record retention) and ISO 27001 Annex A.12.4 (logging and monitoring) requirements.

> **Cross-reference:** `AuditEventConsumer` (this section) subscribes to the same Kafka topics produced by the domain event publishers defined throughout this document. `AuditQueryService.export_for_compliance()` is used by the Compliance Officer role as defined in the CHE RBAC specification.

## Immutability Guarantees

### PostgreSQL Row-Level Security

```sql
-- Enable RLS on the audit table
ALTER TABLE audit_events ENABLE ROW LEVEL SECURITY;

-- Allow audit_writer role to INSERT only
CREATE POLICY audit_insert_only ON audit_events
    FOR INSERT
    TO audit_writer
    WITH CHECK (true);

-- Deny all UPDATE to every role (RESTRICTIVE policies override permissive ones)
CREATE POLICY audit_no_update ON audit_events
    AS RESTRICTIVE
    FOR UPDATE
    TO PUBLIC
    USING (false);

-- Deny all DELETE to every role
CREATE POLICY audit_no_delete ON audit_events
    AS RESTRICTIVE
    FOR DELETE
    TO PUBLIC
    USING (false);

-- Read access for audit_reader role only
CREATE POLICY audit_read ON audit_events
    FOR SELECT
    TO audit_reader
    USING (true);

-- Revoke direct table access from the application role
REVOKE ALL ON audit_events FROM che_app;
GRANT INSERT ON audit_events TO audit_writer;
GRANT SELECT ON audit_events TO audit_reader;
```

### WAL-Level Backup Integrity

PostgreSQL Write-Ahead Log (WAL) is streamed to an immutable object store (e.g. S3 with Object Lock) in `ARCHIVE` mode. Any silent modification of the `audit_events` table would produce a WAL discrepancy detectable during log replay. The `pg_audit` extension additionally logs all DDL operations (including policy changes) to the audit schema, creating a self-referential integrity guarantee.

## AuditEventConsumer

Kafka consumer that subscribes to all CHE domain event topics and writes immutable `AuditEvent` rows. Runs as a dedicated microservice with `audit_writer` database privileges only.

```python

    "margin_pct",
    "profit_margin_pct",
    "target_price_eur",
    "bid_price",
    "bid_price_eur",

    system_source: str

class AuditEventConsumer:
    """
    Kafka consumer that subscribes to all CHE domain event topics and
    writes immutable AuditEvent records to the append-only audit log.

    Topics consumed:
      - che.cost-snapshots.events
      - che.quotes.events
      - che.rfq-rounds.events
      - che.supplier-prices.events
      - che.material-prices.events
      - che.process-costs.events

    CHE_TOPICS: list[str] = [
        "che.cost-snapshots.events",
        "che.quotes.events",
        "che.rfq-rounds.events",
        "che.supplier-prices.events",
        "che.material-prices.events",
        "che.process-costs.events",
    ]

        "CostSnapshotCreated": AuditAction.CREATE,
        "CostSnapshotApproved": AuditAction.APPROVE,
        "CostSnapshotSuperseded": AuditAction.UPDATE,
        "QuoteSubmitted": AuditAction.CREATE,
        "QuoteRevised": AuditAction.UPDATE,
        "QuoteAccepted": AuditAction.APPROVE,
        "QuoteRejected": AuditAction.REJECT,
        "RFQRoundIssued": AuditAction.CREATE,
        "BidReceived": AuditAction.CREATE,
        "BidAwarded": AuditAction.APPROVE,
        "SupplierPriceRecorded": AuditAction.CREATE,
        "SupplierPriceSuperseded": AuditAction.UPDATE,
        "MaterialPriceIndexed": AuditAction.CREATE,
        "ProcessCostRecorded": AuditAction.CREATE,
        "EntityVersionCreated": AuditAction.UPDATE,
        "RetentionPolicyExecuted": AuditAction.ARCHIVE,
    }

        self,
        old_values: Optional[dict],
        new_values: Optional[dict],
    ) -> list[str]:
        if not old_values or not new_values:
            return []
        return sorted(
            k
            for k in set(old_values.keys()) | set(new_values.keys())
            if old_values.get(k) != new_values.get(k)
        )

    def _contains_sensitive_data(self, values: Optional[dict]) -> bool:
        if not values:
            return False
        return bool(SENSITIVE_FIELDS_AUDIT & set(values.keys()))

        Expected raw_event keys:
            event_type, occurred_at, entity_type, entity_id, actor_id,
            actor_role, actor_ip, session_id, correlation_id (optional),
            old_values (optional), new_values (optional),

        old_values: Optional[dict] = raw_event.get("old_values")
        new_values: Optional[dict] = raw_event.get("new_values")

        if action == AuditAction.CREATE and self._contains_sensitive_data(new_values):
            logger.debug(
                "Sensitive fields present in CREATE event for %s %s",
                raw_event["entity_type"],
                raw_event["entity_id"],
            )

        return AuditEventRecord(
            event_id=uuid.uuid4(),
            occurred_at=datetime.fromisoformat(raw_event["occurred_at"]),
            entity_type=raw_event["entity_type"],
            entity_id=uuid.UUID(raw_event["entity_id"]),
            action=action,
            actor_id=uuid.UUID(raw_event["actor_id"]),
            actor_role=raw_event["actor_role"],
            actor_ip=raw_event["actor_ip"],
            session_id=uuid.UUID(raw_event["session_id"]),
            system_source=raw_event.get("system_source", "KAFKA_CONSUMER"),
            correlation_id=(
                uuid.UUID(raw_event["correlation_id"])
                if raw_event.get("correlation_id")
                else None
            ),
            old_values=old_values,
            new_values=new_values,
            changed_fields=changed_fields,
            business_reason=raw_event.get("business_reason"),
        )
```

## Sensitive Access Logging

Any read operation that returns values of `floor_price_eur`, `margin_pct`, `profit_margin_pct`, `bid_price`, or supplier financial data triggers a `READ_SENSITIVE` `AuditEvent`. This is enforced at the **service layer** via the `@log_sensitive_access` decorator applied to repository methods. The decorator emits the audit event to Kafka topic `che.audit.sensitive-access` before returning the result, ensuring no sensitive read can bypass the audit trail.

```python
import functools
from datetime import datetime
from typing import Callable

from .audit_event_consumer import AuditAction  # re-use enum defined above


def log_sensitive_access(field_names: list[str], entity_type: str) -> Callable:
    """
    Decorator that emits a READ_SENSITIVE AuditEvent to Kafka whenever a
    repository method returning sensitive pricing or margin fields is called.

    Usage:
        @log_sensitive_access(["floor_price_eur", "margin_pct"], "QUOTE")
        async def get_quote_full(self, quote_id: UUID, *, audit_context: dict) -> dict:
            ...

    audit_context must contain:
        actor_id (str), actor_role (str), actor_ip (str),
        session_id (str), justification (str | None)
    """
    def decorator(func: Callable) -> Callable:
        @functools.wraps(func)
        async def wrapper(self, *args, audit_context: dict, **kwargs):
            result = await func(self, *args, **kwargs)
            event = {
                "event_type": "SensitiveDataAccessed",
                "occurred_at": datetime.utcnow().isoformat(),
                "entity_type": entity_type,
                "entity_id": str(result.get("id", "")),
                "action": AuditAction.READ_SENSITIVE.value,
                "actor_id": audit_context["actor_id"],
                "actor_role": audit_context["actor_role"],
                "actor_ip": audit_context["actor_ip"],
                "session_id": audit_context["session_id"],
                "system_source": "CHE_API",
                # Values are redacted in the Kafka payload; full values are
                # stored encrypted in FieldChange rows by the audit consumer.
                "new_values": {f: "***REDACTED***" for f in field_names},
                "business_reason": audit_context.get("justification"),
            }
            await self._kafka_producer.send("che.audit.sensitive-access", event)
            return result
        return wrapper
    return decorator
```

```python
    def export_for_compliance(
        self,
        from_dt: datetime,
        to_dt: datetime,
        exported_by: uuid.UUID,
        export_format: str = "JSON",
        entity_types: Optional[list[str]] = None,
    ) -> ComplianceExport:
        """
        Export audit events for regulatory compliance (GDPR, HGB § 257).

        Output rules:
          - RESTRICTED field values are redacted; only field names appear.
          - SHA-256 of the serialised content is computed for integrity verification.
          - Supported formats: "JSON" (structured, human-readable) and "CSV" (tabular).
          - The ComplianceExport itself is recorded as an EXPORT AuditEvent by the caller.
        """
        # Production: query DB with get_changes_in_period() and filter by entity_types
        events: list[AuditEventRecord] = []  # populated by concrete subclass

        export_fields = [
            "event_id", "occurred_at", "entity_type", "entity_id",
            "action", "actor_id", "actor_role", "changed_fields", "business_reason",
        ]

        if export_format == "JSON":
            records = [
                {
                    "event_id": str(e.event_id),
                    "occurred_at": e.occurred_at.isoformat(),
                    "entity_type": e.entity_type,
                    "entity_id": str(e.entity_id),
                    "action": e.action.value,
                    "actor_id": str(e.actor_id),
                    "actor_role": e.actor_role,
                    "changed_fields": e.changed_fields,
                    "business_reason": e.business_reason,
                }
                for e in events
            ]
            content = json.dumps(records, indent=2, ensure_ascii=True).encode("utf-8")

        elif export_format == "CSV":
            buffer = io.StringIO()
            writer = csv.DictWriter(buffer, fieldnames=export_fields)
            writer.writeheader()
            for e in events:
                writer.writerow({
                    "event_id": str(e.event_id),
                    "occurred_at": e.occurred_at.isoformat(),
                    "entity_type": e.entity_type,
                    "entity_id": str(e.entity_id),
                    "action": e.action.value,
                    "actor_id": str(e.actor_id),
                    "actor_role": e.actor_role,

                    "business_reason": e.business_reason or "",
                })
            content = buffer.getvalue().encode("utf-8")

        else:
            raise ValueError(
                f"Unsupported export format '{export_format}'. "
                f"Supported formats: 'JSON', 'CSV'."
            )

        sha256_hash = hashlib.sha256(content).hexdigest()

        return ComplianceExport(
            export_id=uuid.uuid4(),
            entity_type=None,

            period_end=to_dt,
```
